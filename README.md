# Алгоритм #
```sh
1. Программа начинает выполнение.

2. Исходный массив строк вводится пользователем с клавиатуры с помощью метода `AskWords`.

3. Программа подсчитывает количество строк в исходном массиве, длина которых меньше или равна 3 символам, с помощью метода `CreateArray`.

4. Создаётся новый массив размером, равным количеству таких строк.

5. В новый массив добавляются все строки из исходного массива, удовлетворяющие условию длины.

6. Новый массив выводится на экран с помощью метода `ArrayPrint`.

7. Программа завершает выполнение.
```

## Реализация ##
``` sh
// Главный метод программы
void Main()
{
    // Ввод массива строк от пользователя с помощью метода AskWords
    string[] array = AskWords("Введите элементы массива через пробел: ");
    
    // Создание нового массива, состоящего из строк длиной <= 3 символа
    CreateArray(array);
    
    // Вывод нового массива с помощью метода ArrayPrint
    ArrayPrint(CreateArray(array));
}

// Метод для создания нового массива
// Новый массив состоит только из тех строк, длина которых <= 3 символа
string[] CreateArray(string[] array)
{
    // Счётчик для подсчёта строк длиной <= 3 символа
    int count = 0;

    // Перебираем каждую строку в исходном массиве
    foreach (string word in array)
    {
        // Если длина строки меньше или равна 3, увеличиваем счётчик
        if (word.Length <= 3)
        {
            count++;
        }
    }

    // Создаём новый массив нужного размера
    string[] NewArray = new string[count];
    
    // Индекс для заполнения нового массива
    int index = 0;

    // Заполняем новый массив строками, длина которых <= 3
    foreach (string word in array)
    {
        // Если длина строки меньше или равна 3, добавляем её в новый массив
        if (word.Length <= 3)
        {
            NewArray[index] = word;
            index++;
        }
    }
    
    // Возвращаем новый массив
    return NewArray;
}

// Метод для ввода данных от пользователя
// Ввод осуществляется с клавиатуры, строки разделяются пробелами
string[] AskWords(string msg)
{
    // Выводим сообщение пользователю
    System.Console.WriteLine(msg);

    // Вводим строку с клавиатуры и разделяем её на слова по пробелу
    string[] tempArray = Console.ReadLine()!.Split(' ');

    // Возвращаем массив строк
    return tempArray;
}

// Метод для вывода массива строк на экран
void ArrayPrint(string[] array)
{
    // Перебираем каждую строку в массиве
    foreach (string word in array)
    {
        // Выводим строку с пробелом после каждого слова
        System.Console.Write(word + ", ");
    }
}

// Запуск метода Main для начала выполнения программы
Main();
```
